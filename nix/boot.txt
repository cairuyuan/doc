2
使用其他ubuntu启动计算机，或者从硬盘启动
	a  使用easybcd添加一个nerogrub选项
	b  找一个镜像如ubuntu-14.04.1-desktop-i386.iso，提取casper中的initrd.lz和vmlinuz两个文件
	c  三个文件复制到某个盘的根目录
	d  编写引导文件：
#umount -l /isodevice
title Install Ubuntu
root (hd0,0)
kernel (hd0,0)/vmlinuz boot=casper iso-scan/filename=/ubuntu-14.04.1-desktop-i386.iso ro quiet splash locale=zh_CN.UTF-8
initrd (hd0,0)/initrd.lz
#其中hd后面的0表示第1个硬盘，第2个0是三个文件的分区，从0数。后面的文件名要一致
